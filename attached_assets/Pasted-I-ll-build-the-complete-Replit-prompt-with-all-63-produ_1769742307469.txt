// Get appropriate prompt template
function getPromptTemplate(product, brand) {
  const typeTemplates = PROMPT_TEMPLATES[product.type] || PROMPT_TEMPLATES.software;
  const categoryTemplate = typeTemplates[product.category] || typeTemplates[Object.keys(typeTemplates)[0]];
  return categoryTemplate(product, brand);
}

async function scrapeProductPage(product, brand) {
  console.log(`üîç Scraping: ${product.name} from ${product.productPageUrl}`);
  
  let browser;
  try {
    browser = await puppeteer.launch({ 
      headless: 'new',
      args: ['--no-sandbox', '--disable-setuid-sandbox', '--disable-dev-shm-usage']
    });
    
    const page = await browser.newPage();
    await page.setUserAgent('Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36');
    await page.setViewport({ width: 1920, height: 1080 });
    
    await page.goto(product.productPageUrl, { 
      waitUntil: 'networkidle2',
      timeout: 30000 
    });
    
    // Wait for images
    await page.waitForSelector('img', { timeout: 10000 }).catch(() => {});
    await new Promise(r => setTimeout(r, 2000)); // Extra wait for lazy loading
    
    // Extract images with multiple strategies
    const imageData = await page.evaluate((productName, brandName) => {
      const images = [];
      const seen = new Set();
      
      // Strategy 1: Product-specific selectors
      const productSelectors = [
        `img[alt*="${productName}"]`,
        `img[alt*="${brandName}"]`,
        'img[class*="product"]',
        'img[class*="hero"]',
        'img[class*="device"]',
        'img[class*="terminal"]',
        'img[class*="featured"]',
        '.product-image img',
        '.hero-image img',
        '.product-photo img',
        '[data-product] img',
        '.gallery img',
        'main img',
        'article img',
        '.content img'
      ];
      
      for (const selector of productSelectors) {
        try {
          document.querySelectorAll(selector).forEach(img => {
            let src = img.src || img.dataset.src || img.dataset.lazySrc || img.getAttribute('data-original');
            
            if (src && !seen.has(src)) {
              // Skip common non-product images
              const lowerSrc = src.toLowerCase();
              if (lowerSrc.includes('logo') || 
                  lowerSrc.includes('icon') || 
                  lowerSrc.includes('avatar') ||
                  lowerSrc.includes('banner') ||
                  lowerSrc.includes('background') ||
                  lowerSrc.includes('sprite') ||
                  lowerSrc.includes('social') ||
                  lowerSrc.includes('footer') ||
                  lowerSrc.includes('header-bg') ||
                  src.includes('1x1') ||
                  src.includes('pixel')) {
                return;
              }
              
              seen.add(src);
              const rect = img.getBoundingClientRect();
              images.push({
                src: src,
                alt: img.alt || '',
                width: img.naturalWidth || img.width || rect.width,
                height: img.naturalHeight || img.height || rect.height,
                area: (img.naturalWidth || rect.width) * (img.naturalHeight || rect.height)
              });
            }
          });
        } catch (e) {}
      }
      
      // Sort by image area (largest first)
      images.sort((a, b) => b.area - a.area);
      
      return {
        images: images.slice(0, 10),
        pageTitle: document.title
      };
    }, product.name, brand);
    
    await browser.close();
    
    // Filter for reasonably sized images
    const validImages = imageData.images.filter(img => img.width > 150 && img.height > 150);
    
    if (validImages.length > 0) {
      return {
        success: true,
        primaryImage: validImages[0],
        allImages: validImages,
        pageTitle: imageData.pageTitle
      };
    }
    
    return { success: false, error: 'No suitable product images found' };
    
  } catch (error) {
    if (browser) await browser.close();
    console.error(`‚ùå Scrape failed for ${product.name}: ${error.message}`);
    return { success: false, error: error.message };
  }
}

async function downloadImage(url, filepath) {
  try {
    // Handle relative URLs
    if (url.startsWith('//')) url = 'https:' + url;
    
    const response = await axios({
      url: url,
      responseType: 'arraybuffer',
      timeout: 20000,
      headers: {
        'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36',
        'Accept': 'image/*'
      }
    });
    
    await sharp(response.data)
      .png()
      .toFile(filepath);
    
    return { success: true };
  } catch (error) {
    return { success: false, error: error.message };
  }
}

async function generateWithReference(product, brand, referenceImagePath) {
  console.log(`üé® Generating with reference: ${product.name}`);
  
  try {
    const imageBuffer = await fs.readFile(referenceImagePath);
    const base64Image = imageBuffer.toString('base64');
    
    const prompt = getPromptTemplate(product, brand);
    
    const model = genAI.getGenerativeModel({ 
      model: "gemini-2.0-flash-exp",
      generationConfig: {
        responseModalities: ["image", "text"],
      }
    });
    
    const result = await model.generateContent([
      {
        inlineData: {
          mimeType: 'image/png',
          data: base64Image
        }
      },
      { text: prompt }
    ]);
    
    const response = result.response;
    
    for (const part of response.candidates[0].content.parts) {
      if (part.inlineData) {
        return {
          success: true,
          imageData: part.inlineData.data,
          method: 'reference-based'
        };
      }
    }
    
    throw new Error('No image in response');
    
  } catch (error) {
    console.error(`‚ùå Reference generation failed: ${error.message}`);
    return { success: false, error: error.message };
  }
}

async function generateWithoutReference(product, brand) {
  console.log(`üé® Generating without reference: ${product.name}`);
  
  try {
    const prompt = getPromptTemplate(product, brand);
    
    const model = genAI.getGenerativeModel({ 
      model: "gemini-2.0-flash-exp",
      generationConfig: {
        responseModalities: ["image", "text"],
      }
    });
    
    const result = await model.generateContent(prompt);
    const response = result.response;
    
    for (const part of response.candidates[0].content.parts) {
      if (part.inlineData) {
        return {
          success: true,
          imageData: part.inlineData.data,
          method: 'prompt-only'
        };
      }
    }
    
    throw new Error('No image in response');
    
  } catch (error) {
    console.error(`‚ùå Prompt-only generation failed: ${error.message}`);
    return { success: false, error: error.message };
  }
}

async function saveThumbnail(imageData, product) {
  const typeDir = product.type === 'hardware' ? CONFIG.hardwareDir : CONFIG.softwareDir;
  const filename = `${product.id}.png`;
  const filepath = path.join(typeDir, filename);
  
  const buffer = Buffer.from(imageData, 'base64');
  
  await sharp(buffer)
    .resize(CONFIG.thumbnailSize.width, CONFIG.thumbnailSize.height, {
      fit: 'contain',
      background: { r: 255, g: 255, b: 255, alpha: 1 }
    })
    .png()
    .toFile(filepath);
  
  return { filename, filepath };
}

async function processProduct(product, brand) {
  console.log(`\n${'‚îÄ'.repeat(50)}`);
  console.log(`üì¶ ${brand} > ${product.name} (${product.type}/${product.category})`);
  console.log(`${'‚îÄ'.repeat(50)}`);
  
  // Step 1: Try to scrape reference image
  const scrapeResult = await scrapeProductPage(product, brand);
  
  let generationResult;
  
  if (scrapeResult.success && scrapeResult.primaryImage) {
    // Download reference
    const refFilename = `${product.id}-ref.png`;
    const refFilepath = path.join(CONFIG.referenceDir, refFilename);
    
    const downloadResult = await downloadImage(scrapeResult.primaryImage.src, refFilepath);
    
    if (downloadResult.success) {
      console.log(`üì• Reference saved: ${refFilename}`);
      
      // Generate with reference
      generationResult = await generateWithReference(product, brand, refFilepath);
    }
  }
  
  // Fallback to prompt-only if reference failed
  if (!generationResult || !generationResult.success) {
    console.log(`‚ö†Ô∏è Falling back to prompt-only generation`);
    generationResult = await generateWithoutReference(product, brand);
  }
  
  // Save thumbnail
  if (generationResult.success) {
    const saved = await saveThumbnail(generationResult.imageData, product);
    console.log(`‚úÖ Thumbnail saved: ${saved.filename}`);
    
    return {
      product: product.name,
      id: product.id,
      brand: brand,
      type: product.type,
      category: product.category,
      success: true,
      method: generationResult.method,
      filename: saved.filename
    };
  }
  
  return {
    product: product.name,
    id: product.id,
    brand: brand,
    type: product.type,
    category: product.category,
    success: false,
    error: generationResult.error
  };
}

// Load manifest and render thumbnails
fetch('/images/equipiq/manifest.json')
  .then(r => r.json())
  .then(data => {
    data.brands.forEach(brand => {
      const brandSection = document.createElement('div');
      brandSection.className = 'brand-section';
      brandSection.innerHTML = `<h2>${brand.name}</h2>`;
      
      brand.products.forEach(product => {
        const card = `
          <div class="product-card ${product.type}">
            <img src="${product.path}" alt="${product.name}" loading="lazy">
            <h3>${product.name}</h3>
            <span class="category">${product.category}</span>
          </div>
        `;
        brandSection.innerHTML += card;
      });
      
      document.getElementById('equipiq-grid').appendChild(brandSection);
    });
  });
